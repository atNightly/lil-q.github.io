leetcode 207. 课程表

解法一：DFS

class Solution {
    Map<Integer, List<Integer>> map; // 邻接表存边
    int[] visited; // 记录访问情况（-1：当前访问；0：未访问；1：已访问）

    public boolean canFinish(int numCourses, int[][] prerequisites) {
        map = new HashMap<>();
        visited = new int[numCourses];
        for (int i = 0; i < numCourses; i++) {
            map.put(i, new ArrayList<Integer>());
        }
        for (int[] pr : prerequisites) {
            map.get(pr[1]).add(pr[0]);
        }
        for (int i = 0; i < numCourses; i++) {
            if (!backTrack(i)) return false; // 有环立即返回 false
        }
        return true;
    }

    public boolean backTrack(int course) {
        if (visited[course] == -1) return false; // 遇到当前访问的路径上的结点说明有环， 返回 false
        if (visited[course] == 1) return true; // 路径结束，返回 true
        visited[course] = -1; 
        for (int nextCourse : map.get(course)) {
            if (!backTrack(nextCourse)) return false;
        }
        visited[course] = 1;
        return true;
    }
}

解法二：BFS + 入度表

class Solution {
    public boolean canFinish(int numCourses, int[][] prerequisites) {
        // 邻接表存边
        Map<Integer, List<Integer>> map = new HashMap<>();
        for (int i = 0; i < numCourses; i++) {
            map.put(i, new ArrayList<Integer>());
        }
        // 建立入度表
        int[] indegree = new int[numCourses];
        for (int[] pr : prerequisites) {
            map.get(pr[1]).add(pr[0]);
            indegree[pr[0]] += 1;
        }
        // BFS 常用数据结构 Queue
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0) queue.add(i);
        }
        // 不断把入读为 0 的结点加入队列
        while (!queue.isEmpty()) {
            int pre = queue.remove();
            numCourses--;
            for (int cur : map.get(pre)) {
                if (--indegree[cur] == 0) queue.add(cur);
            }
        }
        return numCourses == 0;
    }
}

leetcode 210. 课程表 II

解法一：DFS + 入度表

class Solution {
    List[] edges; // 还可以用 List 来存邻接表，在边较多时效率比 Map 好且不浪费空间
    int[] indegree;
    int[] res; // 全局变量记录 result
    int index = 0; // 全局变量记录 result 当前处理位

    public int[] findOrder(int numCourses, int[][] prerequisites) {
        edges = new ArrayList[numCourses];
        for (int i = 0; i < numCourses; i++) {
            edges[i] = new ArrayList<Integer>();
        }
        indegree = new int[numCourses];
        for (int[] pr : prerequisites) {
            edges[pr[1]].add(pr[0]);
            indegree[pr[0]]++;
        }
        res = new int[numCourses];
        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0) {
                res[index++] = i;
                backTrack(i);
            }
        }
        return index == numCourses ? res : new int[0];
    }

    public void backTrack(int course) {
        indegree[course] = -1; // 入读置为 -1，防止重复运算
        for (int nextCourse : (ArrayList<Integer>) edges[course]) {
            indegree[nextCourse]--;
            if (indegree[nextCourse] == 0) {
                res[index++] = nextCourse;
                backTrack(nextCourse);
            }
        }
    }
}

class Solution {
    public int[] findOrder(int numCourses, int[][] prerequisites) {
        List[] edges = new ArrayList[numCourses];
        for (int i = 0; i < numCourses; i++) {
            edges[i] = new ArrayList<Integer>();
        }
        int[] indegree = new int[numCourses];
        for (int[] pr : prerequisites) {
            edges[pr[1]].add(pr[0]);
            indegree[pr[0]]++;
        }
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < numCourses; i++) {
            if (indegree[i] == 0) queue.add(i);
        }
        int[] res = new int[numCourses];
        int index = 0;
        while (!queue.isEmpty()) {
            int pre = queue.remove();
            res[index++] = pre;
            for (int cur : (ArrayList<Integer>) edges[pre]) {
                if (--indegree[cur] == 0) queue.add(cur);
            }
        }
        return index == numCourses ? res : new int[0];
    }
}
